import sspyrs

# SSRS server details
username = "your_username"
password = "your_password"
report_path = "http://your_server/reportserver?/your_report_folder/your_report_name"
output_file = "path/to/save/report.pdf"

# Report parameters (if any)
params = {'param1': 'value1', 'param2': 'value2'}

try:
    # Initialize the report object
    my_report = sspyrs.report(report_path, username, password, params)

    # Download the report as PDF
    my_report.download('PDF', output_file)
    print(f"Report saved as PDF to {output_file}")

except Exception as e:
    print(f"An error occurred: {e}")
