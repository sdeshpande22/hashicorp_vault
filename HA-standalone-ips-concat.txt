$primary_ns_ips = @()
$secondary_ns_ips = @()
$standalone_ns_ips = @()
$HAdevice_ns_ips = @()

# Hashtable to store the mapping of primary-secondary pairs
$HADevicePairs = @{}

foreach ($Netscaler in $Netscalers) {
    $deviceinfo = (Invoke-RestMethod -Uri "https://$($Global:ADM)/nitro/v2/config/ns?filter=ip_address:$Netscaler" -Method Get -WebSession $admSession.websession -ErrorAction Stop).ns

    if ($deviceinfo.instance_mode -eq "Primary" -and $deviceinfo.is_ha_configured -eq $true) {
        CTX-Logging-DisplayPipelineUpdate -Message "$($deviceinfo.hostname) Primary HA Device Added"
        $primary_ns_ips += $Netscaler

        # Initialize or update the HA pair with primary device
        if (-not $HADevicePairs.ContainsKey($Netscaler)) {
            $HADevicePairs[$Netscaler] = @{}
        }
        $HADevicePairs[$Netscaler]['Primary'] = $Netscaler
    }
    elseif ($deviceinfo.instance_mode -eq "Secondary" -and $deviceinfo.is_ha_configured -eq $true) {
        CTX-Logging-DisplayPipelineUpdate -Message "$($deviceinfo.hostname) Secondary HA Device Added"
        $secondary_ns_ips += $Netscaler

        # Find the corresponding primary (assuming it's already been processed), and pair it
        $primaryKey = $primary_ns_ips | Where-Object { $_ -eq $Netscaler }
        if ($primaryKey) {
            if (-not $HADevicePairs.ContainsKey($primaryKey)) {
                $HADevicePairs[$primaryKey] = @{}
            }
            $HADevicePairs[$primaryKey]['Secondary'] = $Netscaler
        }
    }
    elseif ($deviceinfo.instance_mode -eq "Primary" -and $deviceinfo.is_ha_configured -eq $false) {
        CTX-Logging-DisplayPipelineUpdate -Message "$($deviceinfo.hostname) Standalone Device Added"
        $standalone_ns_ips += $Netscaler
    }
}

# After the loop, combine Primary-Secondary pairs
foreach ($key in $HADevicePairs.Keys) {
    if ($HADevicePairs[$key]['Primary'] -and $HADevicePairs[$key]['Secondary']) {
        $HAdevice_ns_ips += "$($HADevicePairs[$key]['Primary'])-$($HADevicePairs[$key]['Secondary'])"
    }
}

# Output or use the concatenated $HAdevice_ns_ips as needed
$HAdevice_ns_ips
