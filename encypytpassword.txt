import requests
from cryptography.fernet import Fernet

# Define your encryption key
key = b'your_key_here'  # Replace 'your_key_here' with your actual key

# Define a function to decrypt the password
def decrypt_password(encrypted_password):
    cipher = Fernet(key)
    return cipher.decrypt(encrypted_password).decode()

# Example encrypted password (replace this with your actual encrypted password)
encrypted_password = b'your_encrypted_password_here'

# Decrypt the password
password = decrypt_password(encrypted_password)

# Make a request using the decrypted password
url = 'https://example.com/api/endpoint'
response = requests.get(url, auth=('username', password))

# Handle the response
if response.status_code == 200:
    print('Request successful!')
    print('Response:', response.json())
else:
    print('Request failed with status code:', response.status_code)
